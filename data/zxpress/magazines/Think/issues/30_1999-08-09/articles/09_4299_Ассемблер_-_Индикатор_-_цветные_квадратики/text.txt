        LD      A,#08
OUT     (C),A
        IN      A,(C)
        SRL     A
        AND     #07
        SLA     A
        SLA     A
        SLA     A
        LD      (#581E),A
        LD      (#5869),A
        LD      (#5907),A
        LD      (#589D),A
        LD      (#58A6),A
        LD      A,#09
        OUT     (C),A
        IN      A,(C)
        SRL     A
        AND     #07
        SLA     A
        SLA     A
        SLA     A
        LD      (#5823),A
        LD      (#5832),A
LD      (#58E3),A
        LD      (#58BA),A
        LD      (#58C9),A
        LD      A,#0A
        OUT     (C),A
        IN      A,(C)
        SRL     A
        AND     #07
        SLA     A
        SLA     A
        SLA     A
        LD      (#582C),A
        LD      (#5839),A
        LD      (#5875),A
        LD      (#5882),A
