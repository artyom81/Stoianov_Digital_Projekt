;  if (NewLhd.Flags & LHD_PASSWORD)
ld a,(NewLhd_Flags)
and LHD_PASSWORD

;  {

;    {
;      if (GetPassword(1)!=1)
rst #10

;        ErrExit(EEMPTY,USER_BREAK);

;    }
jr EF_nopassword
