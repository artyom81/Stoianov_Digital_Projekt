;+-------------------------------------------------------------+
;│ Ultra-Fast печаталка на б4 символа: Eraser/Delirium Tremens │
;+-------------------------------------------------------------+
;│Вход:  HL    - адрес печати на экране                        │
;│       DE    - адрес расположения данных для печати (б4 byte)│
;│Выход: HL    - адрес след. строчки в curmehte°               │
;+-------------------------------------------------------------+
        ORG #C000
FОNTбЧ
+INCBIN "б4_ТАВ"                                        
                                          
        ORG #б000
Start
LD HL,#4000                                             
        LD DE,SPACE                                             
        CALL PRT_б4                                             
        RET                                                     
                                          
SPACE   DEFB "*************************** FUNeral #01 *********"
        DEFB "***************"
PRT_б4
LD В,#40/#02
Loop_б4
PUSH ВС                                                 
        LD A,(DE)                                               
        CALL PR_б4_L                                            
        INC DE                                                  
        LD A,(DE)                                               
        CALL PR_б4_R                                            
        INC DE                                                  
        POP ВС                                                  
        DJNZ Loop_б4
RET
PR_б4_L
LD В,'FОNTбЧ; Печать в левой части знакоместа           
        LD C,A                                                  
        DUP 7                                                   
        LD A,(ВС)                                               
        AND #F0                                                 
        LD (HL),A                                               
        INC H                                                   
        INC В                                                   
        EDUP                                                    
        LD A,(ВС)                                               
        AND #F0                                                 
        LD (HL),A                                               
        LD A,H                                                  
        SUB #07                                                 
        LD H,A                                                  
        RET
PR_б4_R
LD В,'FОNTбЧ; Печать в правой части знакоместа          
        LD C,A
DUP 7
LD A,(ВС)
AND #0F
OR (HL)
LD (HL),A
INC H
INC В
EDUP
LD A,(ВС)
AND #0F
OR (HL)
LD (HL),A
LD A,H
SUB #07
LD H,A
INC L
RET
;
